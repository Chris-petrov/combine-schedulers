public typealias SchedulerOf<S: Scheduler> = Scheduler<S.SchedulerTimeType>

A convenience to specify a scheduler with the same SchedulerTimeType as a concrete scheduler.

This is most useful for injecting a scheduler into some code, such as a SwiftUI ObservableObject, or Composable Architecture environment. For example, a view model that needs time-based scheduling can accept an any SchedulerOf as a parameter and then use the scheduler to sleep for a duration of time:

class ViewModel: ObservableObject {
  @Published var message: String?

  let mainQueue: any SchedulerOf<DispatchQueue>
  init(mainQueue: any SchedulerOf<DispatchQueue>) {
    self.mainQueue = mainQueue
  }

  func onAppear() {
    Task {
      try await self.mainQueue.sleep(for: .seconds(1))
      self.message = "Welcome!
    }
  }
}
